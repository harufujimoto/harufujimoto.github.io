二分探索O(NlogN)。復元はできない
{% highlight cpp %}

template<class T> int lcs(vector<T>& arr)
{
  int N = arr.size();
  const T INF = numeric_limits<T>::max();
  vector<T> dp(N+1,INF);
  for(int i = 0;i < N;i++){
    *lower_bound(dp.begin(),dp.end(),a[i]) = a[i];
  }
  return (int)(lower_bound(dp.begin(),dp.end(),INF) - dp.begin());
}
